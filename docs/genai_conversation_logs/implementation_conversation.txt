This is a sensible FastAPI + SQLAlchemy layout because it keeps the engine and session factory at module scope, separates ORM models from validation schemas, and lets routers stay thin while business logic lives in services. SQLAlchemy’s docs recommend creating the Engine once and keeping the sessionmaker in the same scope, and FastAPI commonly uses dependency injection for per-request database sessions.

app/core/database.py
from sqlalchemy import create_engine
from sqlalchemy.orm import DeclarativeBase, sessionmaker

from app.core.config import DATABASE_URL

# Needed for SQLite when using FastAPI across requests/threads
engine = create_engine(
    DATABASE_URL,
    connect_args={"check_same_thread": False}
)

SessionLocal = sessionmaker(
    bind=engine,
    autoflush=False,
    autocommit=False,
    expire_on_commit=False
)


class Base(DeclarativeBase):
    pass


def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()
Why this works:

create_engine() is the standard way to create the database engine. 

sessionmaker() is the recommended factory for creating Session objects, usually in the same scope as the engine. 

SQLite support is built into SQLAlchemy, and FastAPI projects commonly use check_same_thread=False so request handling does not trip over SQLite’s default thread restriction. SQLite is fully supported by SQLAlchemy.

from app.schemas.incident import IncidentCreate, IncidentUpdate


def get_all_incidents(db: Session):
    return db.query(Incident).all()


def get_incident_by_id(db: Session, incident_id: int):
    return db.query(Incident).filter(Incident.id == incident_id).first()


def create_incident(db: Session, incident_data: IncidentCreate):
    incident = Incident(**incident_data.model_dump())
    db.add(incident)
    db.commit()
    db.refresh(incident)
    return incident


def update_incident(db: Session, incident: Incident, update_data: IncidentUpdate):
    for field, value in update_data.model_dump(exclude_unset=True).items():
        setattr(incident, field, value)

    db.commit()
    db.refresh(incident)
    return incident


def delete_incident(db: Session, incident: Incident):
    db.delete(incident)
    db.commit()

app/routers/incident_router.py
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session

from app.core.database import get_db
from app.schemas.incident import IncidentCreate, IncidentUpdate, IncidentResponse
from app.services import incident_service

router = APIRouter(prefix="/incidents", tags=["Incidents"])


@router.get("/", response_model=list[IncidentResponse])
def list_incidents(db: Session = Depends(get_db)):
    return incident_service.get_all_incidents(db)


@router.get("/{incident_id}", response_model=IncidentResponse)
def get_incident(incident_id: int, db: Session = Depends(get_db)):
    incident = incident_service.get_incident_by_id(db, incident_id)
    if not incident:
        raise HTTPException(status_code=404, detail="Incident not found")
    return incident


@router.post("/", response_model=IncidentResponse, status_code=201)
def create_incident(payload: IncidentCreate, db: Session = Depends(get_db)):
    return incident_service.create_incident(db, payload)


@router.put("/{incident_id}", response_model=IncidentResponse)
def update_incident(
    incident_id: int,
    payload: IncidentUpdate,
    db: Session = Depends(get_db)
):
    incident = incident_service.get_incident_by_id(db, incident_id)
    if not incident:
        raise HTTPException(status_code=404, detail="Incident not found")

    return incident_service.update_incident(db, incident, payload)


@router.delete("/{incident_id}", status_code=204)
def delete_incident(incident_id: int, db: Session = Depends(get_db)):
    incident = incident_service.get_incident_by_id(db, incident_id)
    if not incident:
        raise HTTPException(status_code=404, detail="Incident not found")

    incident_service.delete_incident(db, incident)
    return None

Thought for 10s
Yes — for this dataset size, the cleanest approach is to read the CSV in chunks, normalize "NULL" to Python None, map each row into plain dictionaries, and then use Session.execute(insert(Booking), batch) for batched inserts. In SQLAlchemy 2.x, the modern bulk approach is Session.execute() with an Insert construct; older bulk helpers like bulk_insert_mappings() are now considered legacy.

import csv
from pathlib import Path
from typing import Any

from sqlalchemy import insert, delete
from sqlalchemy.orm import Session

from app.core.database import SessionLocal, engine, Base
from app.models.booking import Booking


# Adjust this to match your real CSV file location
CSV_PATH = Path("data/hotel_bookings.csv")

# Tune this if needed
BATCH_SIZE = 5000


def nullify(value: str | None) -> str | None:
    """
    Convert CSV placeholders like 'NULL' or empty strings to Python None.
    """
    if value is None:
        return None

    value = value.strip()
    if value == "" or value.upper() == "NULL":
        return None

    return value


def to_int(value: str | None) -> int | None:
    value = nullify(value)
    return int(value) if value is not None else None


def to_float(value: str | None) -> float | None:
    value = nullify(value)
    return float(value) if value is not None else None


def to_bool_int(value: str | None) -> int | None:
    """
    Kaggle hotel dataset uses 0/1 for columns like is_canceled, is_repeated_guest.
    Store as integers if your DB model uses Integer.
    """
    value = nullify(value)
    return int(value) if value is not None else None


def row_to_booking_dict(row: dict[str, str]) -> dict[str, Any]:
    """
    Map CSV columns to your Booking SQLAlchemy model fields.
    Edit this carefully so it matches your actual Booking model.
    """
    return {
        "hotel": nullify(row.get("hotel")),
        "is_canceled": to_bool_int(row.get("is_canceled")),
        "lead_time": to_int(row.get("lead_time")),
        "arrival_date_year": to_int(row.get("arrival_date_year")),
        "arrival_date_month": nullify(row.get("arrival_date_month")),
        "arrival_date_week_number": to_int(row.get("arrival_date_week_number")),
        "arrival_date_day_of_month": to_int(row.get("arrival_date_day_of_month")),
        "stays_in_weekend_nights": to_int(row.get("stays_in_weekend_nights")),
        "stays_in_week_nights": to_int(row.get("stays_in_week_nights")),
        "adults": to_int(row.get("adults")),
        "children": to_float(row.get("children")),  # often float-ish in CSVs with missing values
        "babies": to_int(row.get("babies")),
        "meal": nullify(row.get("meal")),
        "country": nullify(row.get("country")),
        "market_segment": nullify(row.get("market_segment")),
        "distribution_channel": nullify(row.get("distribution_channel")),
        "is_repeated_guest": to_bool_int(row.get("is_repeated_guest")),
        "previous_cancellations": to_int(row.get("previous_cancellations")),
        "previous_bookings_not_canceled": to_int(row.get("previous_bookings_not_canceled")),
        "reserved_room_type": nullify(row.get("reserved_room_type")),
        "assigned_room_type": nullify(row.get("assigned_room_type")),
        "booking_changes": to_int(row.get("booking_changes")),
        "deposit_type": nullify(row.get("deposit_type")),
        "agent": nullify(row.get("agent")),
        "company": nullify(row.get("company")),
        "days_in_waiting_list": to_int(row.get("days_in_waiting_list")),
        "customer_type": nullify(row.get("customer_type")),
        "adr": to_float(row.get("adr")),
        "required_car_parking_spaces": to_int(row.get("required_car_parking_spaces")),
        "total_of_special_requests": to_int(row.get("total_of_special_requests")),
        "reservation_status": nullify(row.get("reservation_status")),
        "reservation_status_date": nullify(row.get("reservation_status_date")),
    }


def chunked_rows(csv_path: Path, batch_size: int):
    """
    Yield lists of mapped booking dictionaries in batches.
    """
    batch: list[dict[str, Any]] = []

    with csv_path.open("r", encoding="utf-8-sig", newline="") as f:
        reader = csv.DictReader(f)

        for row in reader:
            batch.append(row_to_booking_dict(row))

            if len(batch) >= batch_size:
                yield batch
                batch = []

        if batch:
            yield batch


def import_bookings(csv_path: Path, batch_size: int = BATCH_SIZE, clear_existing: bool = False):
    """
    Import bookings in batches using SQLAlchemy bulk INSERT.
    """
    Base.metadata.create_all(bind=engine)

    total_inserted = 0

    with SessionLocal() as db:
        if clear_existing:
            db.execute(delete(Booking))
            db.commit()

        for batch_number, batch in enumerate(chunked_rows(csv_path, batch_size), start=1):
            db.execute(insert(Booking), batch)
            db.commit()
            total_inserted += len(batch)
            print(f"Batch {batch_number}: inserted {len(batch)} rows (total: {total_inserted})")

    print(f"Done. Imported {total_inserted} bookings.")


if __name__ == "__main__":
    import_bookings(CSV_PATH, batch_size=BATCH_SIZE, clear_existing=False)

Why this approach is the right one
Session.execute(insert(Booking), batch) is the modern SQLAlchemy 2.x pattern for ORM bulk inserts, and SQLAlchemy documents batched multi-row insert behavior as the preferred modern route instead of legacy bulk helpers. 

Useful notes for your dataset
The Kaggle hotel dataset often contains missing-like values in fields such as children, country, agent, and company, so converting "NULL" and empty strings to None before insertion will prevent dirty string data leaking into nullable columns. That mapping step is also where you should normalize numeric fields like lead_time and adr into real Python int and float values before insert. This is an implementation recommendation based on your described CSV shape rather than a claim from external docs.

def cancellation_risk_score(booking) -> tuple[int, str, list[str]]:
    score = 0
    reasons = []

    # Lead time
    if booking.lead_time is not None:
        if booking.lead_time <= 7:
            points = 0
        elif booking.lead_time <= 30:
            points = 5
        elif booking.lead_time <= 90:
            points = 12
        elif booking.lead_time <= 180:
            points = 18
        else:
            points = 25
        score += points
        if points:
            reasons.append(f"Long lead time (+{points})")

    # Previous cancellations
    prev = booking.previous_cancellations or 0
    if prev == 0:
        points = 0
    elif prev == 1:
        points = 12
    elif prev == 2:
        points = 18
    else:
        points = 25
    score += points
    if points:
        reasons.append(f"Previous cancellations history (+{points})")

    # Deposit type
    deposit = (booking.deposit_type or "").strip().lower()
    if deposit == "non refund":
        points = 0
    elif deposit == "no deposit":
        points = 10
    elif deposit == "refundable":
        points = 20
    else:
        points = 8
    score += points
    if points:
        reasons.append(f"Deposit type risk (+{points})")

    # Market segment
    segment = (booking.market_segment or "").strip().lower()
    if segment == "corporate":
        points = 0
    elif segment == "direct":
        points = 2
    elif segment in {"offline ta/to"}:
        points = 6
    elif segment in {"online ta", "ta/to"}:
        points = 8
    elif segment == "groups":
        points = 10
    else:
        points = 4
    score += points
    if points:
        reasons.append(f"Market segment pattern (+{points})")

    # Repeat guest
    repeated = booking.is_repeated_guest or 0
    points = 0 if repeated == 1 else 10
    score += points
    if points:
        reasons.append(f"Not a repeat guest (+{points})")

    # Booking changes
    changes = booking.booking_changes or 0
    if changes == 0:
        points = 0
    elif changes == 1:
        points = 3
    elif changes == 2:
        points = 6
    else:
        points = 10
    score += points
    if points:
        reasons.append(f"Multiple booking changes (+{points})")

    score = min(score, 100)

    if score >= 60:
        band = "high"
    elif score >= 30:
        band = "medium"
    else:
        band = "low"

    return score, band, reasons

    SQLALCHEMY_DATABASE_URL,

TestingSessionLocal = sessionmaker(
    autocommit=False,
    autoflush=False,
    bind=engine,
    expire_on_commit=False,

def override_get_db():
    db = TestingSessionLocal()
    try:
        yield db
    finally:
        db.close()

What are the best practices for error handling in a FastAPI REST API? I want to make sure I'm using the correct HTTP status codes (201 for creation, 404 for not found, 422 for validation errors). Should I create custom exception handlers? How do I make sure my error responses are consistent and informative?

I’m checking the current FastAPI and Starlette docs so I can give you a clean set of practices that matches how the framework handles validation and custom exceptions today.

Thought for 10s
Yes — the best pattern in FastAPI is:

use the right HTTP status codes

raise HTTPException for expected API errors

add custom exception handlers for a consistent error shape

keep one shared error response schema across the whole API

let FastAPI handle request validation, but optionally reformat its default 422 output if you want a cleaner contract. FastAPI has built-in handlers for HTTPException and request validation errors, and it lets you override them with @app.exception_handler(...). 

Recommended status codes
For a REST API like your hotel booking system, these are the main ones to use:

200 OK for successful reads and updates that return content.

201 Created for successful creation. FastAPI supports setting this directly on the route with status_code=.... 

204 No Content for successful deletes when you return no body.

400 Bad Request for malformed business requests that are syntactically valid JSON but logically invalid for your domain.

404 Not Found when a booking, incident, or note does not exist.

409 Conflict when the request conflicts with current state, such as trying to create a duplicate unique resource.

422 Unprocessable Content for request validation errors. FastAPI returns this automatically when request data fails validation. 

500 Internal Server Error for unexpected server-side failures; Starlette includes server error handling middleware by default. 

For your examples:

POST /incidents → 201

GET /incidents/999 → 404

invalid request body → 422

DELETE /manager-notes/5 → 204

When to use HTTPException
Use HTTPException for expected application-level failures, like “booking not found” or “incident cannot be created because booking_id is invalid.” FastAPI’s documented error-handling flow is built around raising HTTPException, and FastAPI provides a default JSON response for it. 

Example:

from fastapi import HTTPException, status

def get_booking_or_404(db, booking_id: int):
    booking = db.query(Booking).filter(Booking.id == booking_id).first()
    if not booking:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail=f"Booking {booking_id} not found"
        )
    return booking
Using status.HTTP_404_NOT_FOUND rather than raw integers is a good habit because FastAPI exposes named status constants from Starlette.

class ApiException(Exception):
    def __init__(self, status_code: int, code: str, message: str, details=None):
        self.status_code = status_code
        self.code = code
        self.message = message
        self.details = details


class ResourceNotFound(ApiException):

        super().__init__(
            status_code=404,

            message=f"{resource.title()} {resource_id} was not found"
        )

@app.exception_handler(RequestValidationError)
async def validation_exception_handler(request: Request, exc: RequestValidationError):
    formatted = []
    for err in exc.errors():
        formatted.append({
            "field": ".".join(str(x) for x in err["loc"]),
            "message": err["msg"],
            "type": err["type"],
        })

    return JSONResponse(

                "code": "VALIDATION_ERROR",
                "message": "Request validation failed",
                "details": formatted,

        },

How to keep responses consistent
The easiest way is to standardize on these rules:

every success response uses your normal response models

every error response uses the same top-level error object

every error includes:

a machine-readable code

a human-readable message

optional details

That consistency matters more than making the payload fancy.

For example:

not found:

{
  "error": {
    "code": "INCIDENT_NOT_FOUND",
    "message": "Incident 99 was not found",
    "details": null
  }
}
validation:

{
  "error": {
    "code": "VALIDATION_ERROR",
    "message": "Request validation failed",
    "details": [
      {
        "field": "body.severity",
        "message": "Field required",
        "type": "missing"
      }
    ]
  }
}
Good practical rules for your API
For your hotel booking API, I’d follow these:

Bookings
invalid filter type → 422 if FastAPI/Pydantic catches it

booking not found → 404

Incidents / Manager notes
create success → 201

get missing resource → 404

update missing resource → 404

delete success → 204

invalid body → 422

Analytics / Insights
missing booking for risk score → 404

impossible calculation because of no data:

either return 200 with zeros if that makes business sense

or 404/400 if the endpoint truly requires data
Pick one rule and keep it consistent.

One subtle but important point
Do not use exceptions for normal control flow everywhere. Use them for actual error cases. For example:

“booking not found” → exception is fine

“no rows matched an optional filter” → probably just return [] with 200

That makes your API feel cleaner.
